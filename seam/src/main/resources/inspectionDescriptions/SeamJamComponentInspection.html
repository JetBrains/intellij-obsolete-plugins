Reports configuration issues on annotated Seam classes

Example:


@Name("authenticator")
class SeanComponent {
  @Create // static method not allowed
  private static void create() {...}

   @Destroy // static method not allowed
   private static void destroy() {...}

   @DataModel // not getter
   private Map method() {...}

   @Factory // must be public
   private Map getUniverse2() {...}

   @Unwrap // must have no parameters
   private static Map createUniverse(int days) { ... }
}
